SELECT 
    Prodotti.ProdottoID,
    Prodotti.NomeProdotto,
    Prodotti.Descrizione,
    Prodotti.Prezzo,
    Prodotti.QuantitaDisponibile,
    Prodotti.CategoriaID,
    Prodotti.DataAggiunta,
    Prodotti.ImmagineCoverHome,
    Prodotti.ImmagineDettagli,
    Categorie.NomeCategoria
FROM 
    Prodotti
INNER JOIN 
    Categorie ON Prodotti.CategoriaID = Categorie.CategoriaID
WHERE 
    Categorie.NomeCategoria = 'Elettronica';
